    8. export function TableGridPanel({ spec }: { spec: TableSpecPayload }):
       a. Derivar chaves: columns com key = col.key ?? col.name.toLowerCase().replace(/\s+/g,"_")

       b. Inicializar historyState com useReducer(historyReducer, {
          past: [],
          present: {
            rows: (spec.rows ?? []) as RowData[],
            columns: spec.columns as TableColumn[],
          },
          future: [],
        })

       c. const { displayRows } = useFormulaEngine(
            historyState.present.rows,
            historyState.present.columns,
            spec.separator ?? ";"
          )

       d. const [sortState, setSortState] = useState&lt;{key:string; dir:"asc"|"desc"} | null&gt;(null)

       e. const sortedRows = useMemo(() => {
            if (!sortState) return historyState.present.rows;
            return [...historyState.present.rows].sort((a, b) => {
              const va = a[sortState.key] ?? "";
              const vb = b[sortState.key] ?? "";
              // Usar displayRows para sort de colunas formula
              const cmp = typeof va === "number" && typeof vb === "number"
                ? va - vb
                : String(va).localeCompare(String(vb), "pt-BR");
              return sortState.dir === "asc" ? cmp : -cmp;
            });
          }, [historyState.present.rows, sortState])
          // Pitfall 3: [...rows].sort() — spread antes de sort, nunca mutar

       f. const dsgColumns = useMemo(() => {
            // Pitfall 4: useMemo obrigatório para DynamicDataSheetGrid
            return [
              ...historyState.present.columns.map(col => ({
                ...keyColumn(col.key ?? col.name, textColumn),
                title: (
                  <div className="col-header"
                    data-sort={sortState?.key === col.key ? sortState.dir : undefined}
                    onClick={() => handleSortClick(col.key!)}
                  >
                    {col.name}
                    {sortState?.key === col.key && sortState.dir === "asc" && <ArrowUp size={12} />}
                    {sortState?.key === col.key && sortState.dir === "desc" && <ArrowDown size={12} />}
                    <button className="col-header-remove" type="button"
                      aria-label={`Remover coluna ${col.name}`}
                      onClick={e => { e.stopPropagation(); removeColumn(col.key!); }}
                    >Remover</button>
                  </div>
                ),
                disabled: col.type === "formula" ? () => true : undefined,
                // Colunas formula são read-only (D-06, UI-SPEC §TAB-01)
              })),
              // stickyRightColumn para remoção de linha (TAB-03)
              {
                ...keyColumn("__delete__", textColumn),
                title: "",
                width: 36,
                component: ({ rowIndex }: { rowIndex: number }) => (
                  <button type="button" aria-label={`Remover linha ${rowIndex + 1}`}
                    onClick={() => removeRow(rowIndex)}
                    style={{ background:"none", border:0, cursor:"pointer", color:"var(--muted)" }}
                  ><X size={14} /></button>
                ),
                disabled: () => true,
              }
            ];
          }, [historyState.present.columns, sortState])

       g. Handlers (useCallback):
          - handleChange = useCallback((newRows) => {
              dispatch({ type:"SET", newState: { rows: newRows, columns: historyState.present.columns } })
            }, [historyState.present.columns])

          - handleSortClick = useCallback((key: string) => {
              setSortState(prev => {
                if (prev?.key !== key) return { key, dir:"asc" };
                if (prev.dir === "asc") return { key, dir:"desc" };
                return null;
              });
            }, [])

          - addRow = useCallback(() => {
              if (historyState.present.rows.length >= 200) return; // TAB-06
              const newRow: RowData = {};
              historyState.present.columns.forEach(c => { newRow[c.key!] = ""; });
              dispatch({ type:"SET", newState: {
                rows: [...historyState.present.rows, newRow],
                columns: historyState.present.columns,
              }});
            }, [historyState.present])

          - addColumn = useCallback(() => {
              if (historyState.present.columns.length >= 26) return; // TAB-06
              const newKey = `coluna_${Date.now()}`;
              dispatch({ type:"SET", newState: {
                rows: historyState.present.rows.map(r => ({ ...r, [newKey]: "" })),
                columns: [...historyState.present.columns, { name:"Nova Coluna", type:"text", key:newKey }],
              }});
            }, [historyState.present])

          - removeColumn = useCallback((key: string) => {
              dispatch({ type:"SET", newState: {
                rows: historyState.present.rows.map(r => { const {[key]:_, ...rest} = r; return rest; }),
                columns: historyState.present.columns.filter(c => c.key !== key),
              }});
            }, [historyState.present])

          - removeRow = useCallback((index: number) => {
              dispatch({ type:"SET", newState: {
                rows: historyState.present.rows.filter((_,i) => i !== index),
                columns: historyState.present.columns,
              }});
            }, [historyState.present])

       h. useEffect para Ctrl+Z / Ctrl+Y (TAB-04):
